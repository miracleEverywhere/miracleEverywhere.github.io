function install_dmp() {
	check_jq
	check_curl

	echo_cyan "æ­£åœ¨è·å–ä¸‹è½½ä¿¡æ¯"

		exit 1

		echo_red "è·å–æœ€æ–°ç‰ˆæœ¬ä¿¡æ¯å¤±è´¥ï¼Œè¯·æ£€æŸ¥ç½‘ç»œè¿æ¥"
		exit 1

	echo_cyan "æ­£åœ¨ä»${url}è¿›è¡Œä¸‹è½½"
	if download "${url}" "dmp.tgz" 10; then

			exit 1
		fi
	else
		echo_red "DMPä¸‹è½½å¤±è´¥"
		exit 1
